// Helper to ensure the "Dr." title exists in Arabic function ensureDoctorTitle(name) { const trimmed = name.trim(); if (!trimmed) return ""; // Checks if it starts with "د." or "د " if (trimmed.startsWith("د.") || trimmed.startsWith("د ")) { return trimmed; } return "د. " + trimmed; } function addResident() { const nameEl = document.getElementById('new-name'); const phoneEl = document.getElementById('new-phone'); const rawName = nameEl.value.trim(); if (!rawName) return; // Apply the Title Guard 🇮🇶 const finalName = ensureDoctorTitle(rawName); db.names.push({ name: finalName, spec: document.getElementById('new-spec').value, phone: formatIraqiPhone(phoneEl.value.trim()) }); nameEl.value = ""; phoneEl.value = ""; renderResidents(); } function saveEdit() { if (currentEditIdx === null) return; const rawName = document.getElementById('edit-name').value.trim(); db.names[currentEditIdx] = { name: ensureDoctorTitle(rawName), // Apply title guard here too phone: formatIraqiPhone(document.getElementById('edit-phone').value.trim()), spec: document.getElementById('edit-spec').value }; closeModal(); renderResidents(); }
